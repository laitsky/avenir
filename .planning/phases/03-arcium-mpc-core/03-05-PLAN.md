```typescript
// Benchmark methodology (Claude's discretion per CONTEXT.md):
// - N=10 sequential update_pool calls (measure each individually)
// - Measure 3 phases per call:
//   Phase A: Client encryption time (encrypt -> create tx)
//   Phase B: Transaction submission time (submit tx -> confirmed)
//   Phase C: MPC computation time (submitted -> callback finalized)
//   Total = A + B + C (this is what the user experiences)
// - Record min, max, mean, median, p95 for each phase and total
// - Run on devnet with real Arx nodes

const results: BenchmarkResult[] = [];

for (let i = 0; i < NUM_RUNS; i++) {
    const startEncrypt = performance.now();
    const encrypted = encryptBetInput(mxePublicKey, i % 2 === 0, 1_000_000);
    const encryptTime = performance.now() - startEncrypt;

    const startSubmit = performance.now();
    await program.methods.updatePool(...).rpc({ skipPreflight: true, commitment: "confirmed" });
    const submitTime = performance.now() - startSubmit;

    const startMpc = performance.now();
    await awaitComputationFinalization(provider, encrypted.computationOffset, programId, "confirmed");
    const mpcTime = performance.now() - startMpc;

    results.push({
        run: i + 1,
        encryptMs: encryptTime,
        submitMs: submitTime,
        mpcMs: mpcTime,
        totalMs: encryptTime + submitTime + mpcTime,
    });

    // Wait for mpc_lock to clear before next iteration
}

// Print results table
// Calculate statistics: min, max, mean, median, p95
```

**Step 4: Run benchmark**
Execute the benchmark script against devnet. Record results.

NOTE: Sequential lock means we must wait for each MPC call to complete before the next. This is by design -- we're measuring worst-case sequential latency.

NOTE: If devnet is unavailable or deployment fails, document the failure and use local test results as a baseline (with clear caveat that local results are not representative).
